Comandos:

Sequencia para subir arquivos:

git add
git commit -m "mensagem"
git push

//criar repositorio
git init

//clonagem de repositorio.
git clone <url>

//adicionar arquivo na stagearea
git add <arquivo>

//commit com mesnagem
git commit -m "<sua mensagem>"

//ver commits de cima para baixo
git log

//mandar commit para repositorio
git push

//deletar arquivo untracked
git clean <arquivco>

//ver diferenca do codigo novo para o antigo
gitk <arquivo>

//passar arquivos para unstage
git restore --stage <file>

//ver variavel origin
git remote -v

//listar branch
git branch --list ou git branch

//ajuda nos comandos
git <comando> --help

//ver branchs romote e local
git branch -a

//baixar branch remote
git checkout -t origin/<branchname>

//excluir branch remotamente
git push --delete origin <branchname>
